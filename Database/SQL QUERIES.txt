SQL QUERIES

SQL TOP, LIMIT, FETCH FIRST Clause

SELECT TOP 10 col_1, col_2 FROM table_name; -- Fetches the first 10 rows



SQL SELECT FIRST

SELECT TOP 1 col_1, col_2 FROM table_name
ORDER BY col_date ASC; -- Fetches the first row based on ascending order of col_date

SQL SELECT LAST

SELECT TOP 1 col_1, col_2 FROM table_name
ORDER BY col_date DESC; -- Fetches the last row based on descending order of col_date

SQL SELECT RANDOM

SELECT TOP 1 col_1, col_2 FROM table_name
ORDER BY NEWID(); -- Fetches one random row

SQL SELECT IN Statement

SELECT col_1, col_2 FROM table_name
WHERE col_category IN ('value_1', 'value_2', 'value_3'); -- Fetches rows where col_category matches specific values

SQL SELECT DATE

SELECT col_date FROM table_name
WHERE col_date > '2024-01-01'; -- Fetches rows where col_date is after a specific date

SQL Query to Insert Multiple Rows

INSERT INTO table_name (col_1, col_2, col_3)
VALUES ('value_1', 'value_2', 'value_3'),
       ('value_4', 'value_5', 'value_6'),
       ('value_7', 'value_8', 'value_9'); -- Inserts multiple rows

SQL INSERT INTO Statement

INSERT INTO table_name (col_1, col_2, col_3)
VALUES ('value_1', 'value_2', 'value_3'); -- Inserts a single row

SQL UPDATE Statement

UPDATE table_name
SET col_1 = 'new_value_1', col_2 = 'new_value_2'
WHERE col_id = 1; -- Updates specific rows

SQL DELETE Statement

DELETE FROM table_name
WHERE col_id = 1; -- Deletes rows based on a condition

SQL Query to Delete Duplicate Rows

WITH CTE AS (
    SELECT col_1, col_2, ROW_NUMBER() OVER (PARTITION BY col_1, col_2 ORDER BY col_id) AS row_num
    FROM table_name
)
DELETE FROM CTE WHERE row_num > 1; -- Deletes duplicate rows while keeping one instance

SQL CLAUSES

SQL WHERE Clause

SELECT col_1, col_2 FROM table_name
WHERE col_3 = 'value_1' AND col_4 > 100; -- Filters rows based on specific conditions

SQL WITH Clause

WITH cte_name AS (
    SELECT col_1, col_2, col_3 FROM table_name
    WHERE col_4 > 100
)
SELECT col_1, col_3 FROM cte_name
WHERE col_3 = 'value_1'; -- Uses a common table expression (CTE) for better readability

SQL HAVING Clause with Examples

SELECT col_1, COUNT(col_2) AS count_col FROM table_name
GROUP BY col_1
HAVING COUNT(col_2) > 10; -- Filters groups based on aggregate function conditions

SQL ORDER BY

SELECT col_1, col_2 FROM table_name
ORDER BY col_3 ASC, col_4 DESC; -- Sorts rows based on multiple columns

SQL GROUP BY

SELECT col_1, COUNT(col_2) AS count_col FROM table_name
GROUP BY col_1; -- Groups rows and calculates aggregate values for each group

SQL LIMIT Clause (For MSSQL, Use TOP)

SELECT TOP 5 col_1, col_2 FROM table_name
ORDER BY col_3 DESC; -- Fetches the top 5 rows based on a specific order

SQL OPERATORS

SQL AND and OR Operators

SELECT col_1, col_2 FROM table_name
WHERE col_3 = 'value_1' AND col_4 > 100 OR col_5 < 50; -- Combines conditions using AND/OR

SQL LIKE Operator

SELECT col_1, col_2 FROM table_name
WHERE col_3 LIKE '%value%'; -- Filters rows where col_3 contains 'value'

SQL IN Operator

SELECT col_1, col_2 FROM table_name
WHERE col_3 IN ('value_1', 'value_2', 'value_3'); -- Filters rows matching a list of values

SQL NOT Operator

SELECT col_1, col_2 FROM table_name
WHERE NOT col_3 = 'value_1'; -- Filters rows where col_3 is NOT 'value_1'

SQL NOT EQUAL Operator

SELECT col_1, col_2 FROM table_name
WHERE col_3 != 'value_1'; -- Filters rows where col_3 is not equal to 'value_1'

SQL IS NULL

SELECT col_1, col_2 FROM table_name
WHERE col_3 IS NULL; -- Filters rows where col_3 has NULL values

SQL UNION Operator

SELECT col_1, col_2 FROM table_name_1
UNION
SELECT col_1, col_2 FROM table_name_2; -- Combines rows from two queries (distinct results)

SQL UNION ALL

SELECT col_1, col_2 FROM table_name_1
UNION ALL
SELECT col_1, col_2 FROM table_name_2; -- Combines rows from two queries (allows duplicates)

SQL Except Clause

SELECT col_1, col_2 FROM table_name_1
EXCEPT
SELECT col_1, col_2 FROM table_name_2; -- Returns rows from the first query that are not in the second query

SQL BETWEEN Operator

SELECT col_1, col_2 FROM table_name
WHERE col_3 BETWEEN 10 AND 20; -- Filters rows where col_3 is within the range 10 to 20

SQL ALL and ANY

-- ALL
SELECT col_1, col_2 FROM table_name
WHERE col_3 > ALL (SELECT col_3 FROM another_table); -- Filters rows where col_3 is greater than all values in the subquery

-- ANY
SELECT col_1, col_2 FROM table_name
WHERE col_3 > ANY (SELECT col_3 FROM another_table); -- Filters rows where col_3 is greater than any value in the subquery

SQL INTERSECT Clause

SELECT col_1, col_2 FROM table_name_1
INTERSECT
SELECT col_1, col_2 FROM table_name_2; -- Returns rows common to both queries

SQL EXISTS

SELECT col_1, col_2 FROM table_name
WHERE EXISTS (
    SELECT 1 FROM another_table
    WHERE another_table.col_3 = table_name.col_3
); -- Filters rows if the subquery returns any rows

SQL CASE Statement

SELECT col_1, 
    CASE 
        WHEN col_2 > 100 THEN 'High'
        WHEN col_2 BETWEEN 50 AND 100 THEN 'Medium'
        ELSE 'Low'
    END AS col_category
FROM table_name; -- Adds a new column with conditional logic

SQL AGGREGATE FUNCTION

SQL COUNT(), AVG(), and SUM() Function

-- COUNT: Counts the number of rows
SELECT COUNT(*) AS total_records FROM table_name;

-- AVG: Calculates the average of a numeric column
SELECT AVG(col_1) AS average_value FROM table_name;

-- SUM: Calculates the total sum of a numeric column
SELECT SUM(col_1) AS total_sum FROM table_name;


SQL MIN() and MAX() Functions

-- MIN: Finds the smallest value in a column
SELECT MIN(col_1) AS minimum_value FROM table_name;

-- MAX: Finds the largest value in a column
SELECT MAX(col_1) AS maximum_value FROM table_name;

Combining COUNT(), AVG(), and SUM()

SELECT 
    COUNT(*) AS total_records,
    AVG(col_1) AS average_value,
    SUM(col_1) AS total_sum
FROM table_name;

Combining MIN() and MAX()

SELECT 
    MIN(col_1) AS minimum_value,
    MAX(col_1) AS maximum_value
FROM table_name;

Combining COUNT(), AVG(), SUM(), MIN(), and MAX()

SELECT 
    COUNT(*) AS total_records,
    AVG(col_1) AS average_value,
    SUM(col_1) AS total_sum,
    MIN(col_1) AS minimum_value,
    MAX(col_1) AS maximum_value
FROM table_name;

SQL Joins (Inner, Left, Right, and Full Join)

-- Example: INNER JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
INNER JOIN table_2 t2 
ON t1.id = t2.foreign_id;

-- Example: LEFT JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
LEFT JOIN table_2 t2 
ON t1.id = t2.foreign_id;

-- Example: RIGHT JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
RIGHT JOIN table_2 t2 
ON t1.id = t2.foreign_id;

-- Example: FULL JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
FULL JOIN table_2 t2 
ON t1.id = t2.foreign_id;


SQL Outer Join

-- FULL OUTER JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
FULL OUTER JOIN table_2 t2 
ON t1.id = t2.foreign_id;


SQL LEFT JOIN

-- Fetches all records from table_1 and matching records from table_2
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
LEFT JOIN table_2 t2 
ON t1.id = t2.foreign_id;

SQL RIGHT JOIN

-- Fetches all records from table_2 and matching records from table_1
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
RIGHT JOIN table_2 t2 
ON t1.id = t2.foreign_id;

SQL FULL JOIN

-- Combines results of both LEFT and RIGHT JOIN
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
FULL JOIN table_2 t2 
ON t1.id = t2.foreign_id;

SQL CROSS JOIN

-- Creates a Cartesian product of the two tables
SELECT t1.col_1, t2.col_2 
FROM table_1 t1 
CROSS JOIN table_2 t2;

SQL Self Join

-- Joining a table to itself
SELECT a.col_1, b.col_2 
FROM table_name a 
INNER JOIN table_name b 
ON a.parent_id = b.id;

SQL | UPDATE with JOIN

-- Updating records in one table based on another table
UPDATE t1
SET t1.col_1 = t2.col_2
FROM table_1 t1 
INNER JOIN table_2 t2 
ON t1.id = t2.foreign_id;

SQL DELETE JOIN

-- Deleting records based on a join condition
DELETE t1
FROM table_1 t1 
INNER JOIN table_2 t2 
ON t1.id = t2.foreign_id;

Recursive Join in SQL

-- Using a Common Table Expression (CTE) for recursion
WITH RecursiveCTE AS (
    SELECT col_1, parent_id
    FROM table_name
    WHERE parent_id IS NULL
    UNION ALL
    SELECT t.col_1, t.parent_id
    FROM table_name t
    INNER JOIN RecursiveCTE r
    ON t.parent_id = r.col_1
)
SELECT * FROM RecursiveCTE;
